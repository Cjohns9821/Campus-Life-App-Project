/* ------------------------------------------------------------ ANNOUNCEMENTS PAGE: Load Announcements ------------------------------------------------------------ */ const announcementsBtn = document.getElementById("loadAnnouncementsBtn"); if (announcementsBtn) { announcementsBtn.addEventListener("click", () => { console.log("Announcements button clicked"); loadAnnouncements(); }); } function loadAnnouncements() { const apiURL = "https://api.quotable.io/quotes?limit=3"; // NEW API URL console.log("Fetching announcements from:", apiURL); fetch(apiURL) .then(response => response.json()) .then(data => { const announcementsSection = document.getElementById("announcementsSection"); // Quotable API returns { results: [...] } const items = data.results; announcementsSection.innerHTML = items .map(item => `
${item.author}

${item.content}

`) .join(""); }) .catch(error => { console.warn("API error. Showing placeholder.", error); const announcementsSection = document.getElementById("announcementsSection"); announcementsSection.innerHTML = `
Announcements Placeholder: Live announcements could not be loaded.
`; }); }
